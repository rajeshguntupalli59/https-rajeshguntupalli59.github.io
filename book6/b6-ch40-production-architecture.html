Chapter 40: Architecture Reference: Production AI App on PostgreSQL

This final chapter brings everything together. Across the preceding thirty-nine chapters, you built individual pieces: embedding pipelines, conversation memory, agent tool registries, RAG retrievers, audit logs, async ingestion workers, caching layers, and production hardening patterns. Now you assemble them into a single coherent reference architecture — a complete, production-grade AI application where PostgreSQL is not just the database but the central nervous system. This chapter is a working blueprint you can fork, adapt, and deploy. Every schema, every API endpoint, every background worker shown here has been drawn from patterns that run in production systems today.

---

40.1 System Overview and Design Philosophy

Before writing a single line of code, the architecture must answer three questions: Where does state live? Where does intelligence live? Where does trust live?

In most AI-native applications, these answers get muddled. State is scattered across Redis, S3, Pinecone, and a relational database with no clear authority. Intelligence is a black box API call with no observability. Trust is an afterthought bolted on after launch.

The architecture in this chapter answers them cleanly:

  • → **State lives in PostgreSQL.** Conversations, embeddings, documents, tool results, audit logs, user permissions — all of it. PostgreSQL becomes the single source of truth.
  • → **Intelligence lives at the boundary.** LLM APIs (Claude, OpenAI) are called from application code and agent workers. Their inputs and outputs are always recorded in PostgreSQL before any side effect is applied.
  • → **Trust lives in the database.** Row-level security, role separation, and cryptographic audit trails enforce access control at the data layer, not just the API layer.

    • The application we are building is an enterprise knowledge assistant: users ask questions, the system retrieves relevant documents via RAG, optionally invokes tools (database lookups, web search, code execution), maintains multi-turn conversation history, and generates responses with full citation. Everything is observable, auditable, and recoverable.

    Component inventory:

    | Layer | Technology | Role |

    |---|---|---|

    | API | FastAPI | HTTP endpoints, streaming responses |

    | Database | PostgreSQL 16 + pgvector | All persistent state |

    | Connection pool | PgBouncer | Connection management |

    | Cache | Redis | Embedding cache, session tokens |

    | Async workers | Celery + Redis | Document ingestion, re-embedding |

    | LLM | Claude 3.5 Sonnet / GPT-4o | Generation and reasoning |

    | Embeddings | OpenAI text-embedding-3-small | Vector representations |

    | Orchestration | LangChain / LlamaIndex | RAG pipelines, agent loops |

    | Protocol | MCP (Model Context Protocol) | Tool exposure to agents |

    | Replication | Logical replication | Read replicas for embedding queries |

-- ============================================================
-- TENANCY AND USERS
-- ============================================================
CREATE TABLE ai.tenants (
    tenant_id     UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name          TEXT NOT NULL,
    plan          TEXT NOT NULL DEFAULT 'standard' CHECK (plan IN ('standard', 'pro', 'enterprise')),
    settings      JSONB NOT NULL DEFAULT '{}',
    created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
    is_active     BOOLEAN NOT NULL DEFAULT true
);

CREATE TABLE ai.users (
    user_id       UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id     UUID NOT NULL REFERENCES ai.tenants(tenant_id) ON DELETE CASCADE,
    email         TEXT NOT NULL,
    display_name  TEXT,
    role          TEXT NOT NULL DEFAULT 'member' CHECK (role IN ('admin', 'member', 'viewer')),
    settings      JSONB NOT NULL DEFAULT '{}',
    created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
    last_active   TIMESTAMPTZ,
    is_active     BOOLEAN NOT NULL DEFAULT true,
    UNIQUE (tenant_id, email)
);

CREATE INDEX idx_users_tenant ON ai.users(tenant_id);

-- ============================================================
-- DOCUMENT CORPUS
-- ============================================================
CREATE TABLE ai.document_collections (
    collection_id   UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID NOT NULL REFERENCES ai.tenants(tenant_id) ON DELETE CASCADE,
    name            TEXT NOT NULL,
    description     TEXT,
    embedding_model TEXT NOT NULL DEFAULT 'text-embedding-3-small',
    chunk_strategy  JSONB NOT NULL DEFAULT '{"type": "recursive", "chunk_size": 512, "overlap": 64}',
    metadata        JSONB NOT NULL DEFAULT '{}',
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE ai.documents (
    document_id     UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    collection_id   UUID NOT NULL REFERENCES ai.document_collections(collection_id) ON DELETE CASCADE,
    tenant_id       UUID NOT NULL REFERENCES ai.tenants(tenant_id) ON DELETE CASCADE,
    title           TEXT NOT NULL,
    source_uri      TEXT,
    source_type     TEXT NOT NULL CHECK (source_type IN ('pdf', 'web', 'markdown', 'plain_text', 'database', 'api')),
    content_hash    TEXT NOT NULL,  -- SHA-256 of raw content, dedup guard
    raw_content     TEXT,           -- optional, for re-chunking
    metadata        JSONB NOT NULL DEFAULT '{}',
    word_count      INTEGER,
    language        TEXT DEFAULT 'en',
    status          TEXT NOT NULL DEFAULT 'pending'
                        CHECK (status IN ('pending', 'processing', 'ready', 'failed', 'archived')),
    ingested_by     UUID REFERENCES ai.users(user_id),
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    UNIQUE (collection_id, content_hash)
);

CREATE INDEX idx_documents_collection ON ai.documents(collection_id);
CREATE INDEX idx_documents_status ON ai.documents(status);
CREATE INDEX idx_documents_tenant ON ai.documents(tenant_id);
CREATE INDEX idx_documents_metadata ON ai.documents USING gin(metadata);

CREATE TABLE ai.document_chunks (
    chunk_id        UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    document_id     UUID NOT NULL REFERENCES ai.documents(document_id) ON DELETE CASCADE,
    collection_id   UUID NOT NULL REFERENCES ai.document_collections(collection_id) ON DELETE CASCADE,
    tenant_id       UUID NOT NULL REFERENCES ai.tenants(tenant_id) ON DELETE CASCADE,
    chunk_index     INTEGER NOT NULL,
    content         TEXT NOT NULL,
    token_count     INTEGER,
    embedding       vector(1536),   -- text-embedding-3-small dimension
    embedding_model TEXT NOT NULL DEFAULT 'text-embedding-3-small',
    embedding_at    TIMESTAMPTZ,
    metadata        JSONB NOT NULL DEFAULT '{}',  -- page number, section, headers
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    UNIQUE (document_id, chunk_index)
);

-- HNSW index for fast approximate nearest neighbor search
CREATE INDEX idx_chunks_embedding ON ai.document_chunks
    USING hnsw (embedding vector_cosine_ops)
    WITH (m = 16, ef_construction = 64);

CREATE INDEX idx_chunks_document ON ai.document_chunks(document_id);
CREATE INDEX idx_chunks_collection ON ai.document_chunks(collection_id);
CREATE INDEX idx_chunks_tenant ON ai.document_chunks(tenant_id);
CREATE INDEX idx_chunks_metadata ON ai.document_chunks USING gin(metadata);

-- ============================================================
-- CONVERSATION MANAGEMENT
-- ============================================================
CREATE TABLE ai.conversations (
    conversation_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID NOT NULL REFERENCES ai.tenants(tenant_id) ON DELETE CASCADE,
    user_id         UUID NOT NULL REFERENCES ai.users(user_id) ON DELETE CASCADE,
    title           TEXT,
    collection_id   UUID REFERENCES ai.document_collections(collection_id),
    agent_config    JSONB NOT NULL DEFAULT '{}',  -- which tools, model, temperature
    summary         TEXT,          -- rolling summary for long conversations
    summary_at      TIMESTAMPTZ,   -- when summary was last generated
    message_count   INTEGER NOT NULL DEFAULT 0,
    token_usage     JSONB NOT NULL DEFAULT '{"input": 0, "output": 0}',
    status          TEXT NOT NULL DEFAULT 'active' CHECK (status IN ('active', 'archived', 'deleted')),
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_conversations_user ON ai.conversations(user_id);
CREATE INDEX idx_conversations_tenant ON ai.conversations(tenant_id);
CREATE INDEX idx_conversations_updated ON ai.conversations(updated_at DESC);

CREATE TABLE ai.messages (
    message_id      UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    conversation_id UUID NOT NULL REFERENCES ai.conversations(conversation_id) ON DELETE CASCADE,
    tenant_id       UUID NOT NULL REFERENCES ai.tenants(tenant_id) ON DELETE CASCADE,
    role            TEXT NOT NULL CHECK (role IN ('user', 'assistant', 'system', 'tool')),
    content         TEXT NOT NULL,
    content_parts   JSONB,         -- for multi-modal: [{type: "text"}, {type: "image_url"}]
    tool_call_id    TEXT,          -- links tool responses to tool calls
    tool_name       TEXT,          -- which tool was called
    tool_input      JSONB,         -- tool call arguments
    tool_output     JSONB,         -- tool call result
    model           TEXT,          -- which LLM generated this
    usage           JSONB,         -- {input_tokens, output_tokens, cache_read_tokens}
    latency_ms      INTEGER,       -- time to first token
    finish_reason   TEXT,          -- stop, length, tool_use, error
    citations       JSONB,         -- [{chunk_id, score, excerpt}]
    sequence_num    INTEGER NOT NULL,  -- ordering within conversation
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_messages_conversation ON ai.messages(conversation_id, sequence_num);
CREATE INDEX idx_messages_tenant ON ai.messages(tenant_id);
CREATE INDEX idx_messages_role ON ai.messages(conversation_id, role);

-- Trigger to auto-increment sequence_num within a conversation
CREATE OR REPLACE FUNCTION ai.next_message_sequence()
RETURNS TRIGGER AS $$
BEGIN
    SELECT COALESCE(MAX(sequence_num), 0) + 1
    INTO NEW.sequence_num
    FROM ai.messages
    WHERE conversation_id = NEW.conversation_id;

    -- Update conversation counters
    UPDATE ai.conversations
    SET message_count = message_count + 1,
        updated_at = now(),
        token_usage = token_usage || jsonb_build_object(
            'input',  (token_usage->>'input')::int + COALESCE((NEW.usage->>'input_tokens')::int, 0),
            'output', (token_usage->>'output')::int + COALESCE((NEW.usage->>'output_tokens')::int, 0)
        )
    WHERE conversation_id = NEW.conversation_id;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_message_sequence
    BEFORE INSERT ON ai.messages
    FOR EACH ROW EXECUTE FUNCTION ai.next_message_sequence();

-- ============================================================
-- AGENT TOOL REGISTRY
-- ============================================================
CREATE TABLE ai.tools (
    tool_id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID REFERENCES ai.tenants(tenant_id) ON DELETE CASCADE,  -- NULL = global tool
    name            TEXT NOT NULL,
    version         TEXT NOT NULL DEFAULT '1.0.0',
    description     TEXT NOT NULL,
    input_schema    JSONB NOT NULL,   -- JSON Schema for tool parameters
    output_schema   JSONB,
    tool_type       TEXT NOT NULL CHECK (tool_type IN ('function', 'retrieval', 'action', 'mcp')),
    mcp_server_uri  TEXT,             -- for MCP-backed tools
    endpoint        TEXT,             -- for HTTP tools
    is_enabled      BOOLEAN NOT NULL DEFAULT true,
    requires_approval BOOLEAN NOT NULL DEFAULT false,
    metadata        JSONB NOT NULL DEFAULT '{}',
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    UNIQUE (COALESCE(tenant_id::text, 'global'), name, version)
);

CREATE TABLE ai.tool_executions (
    execution_id    UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    message_id      UUID REFERENCES ai.messages(message_id),
    tool_id         UUID NOT NULL REFERENCES ai.tools(tool_id),
    conversation_id UUID NOT NULL REFERENCES ai.conversations(conversation_id),
    tenant_id       UUID NOT NULL REFERENCES ai.tenants(tenant_id),
    user_id         UUID NOT NULL REFERENCES ai.users(user_id),
    input           JSONB NOT NULL,
    output          JSONB,
    error           TEXT,
    status          TEXT NOT NULL DEFAULT 'pending'
                        CHECK (status IN ('pending', 'running', 'success', 'failed', 'cancelled')),
    approved_by     UUID REFERENCES ai.users(user_id),
    approved_at     TIMESTAMPTZ,
    started_at      TIMESTAMPTZ,
    completed_at    TIMESTAMPTZ,
    latency_ms      INTEGER,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_tool_exec_conversation ON ai.tool_executions(conversation_id);
CREATE INDEX idx_tool_exec_status ON ai.tool_executions(status);
CREATE INDEX idx_tool_exec_tenant ON ai.tool_executions(tenant_id);

-- ============================================================
-- AUDIT LOGGING
-- ============================================================
CREATE TABLE audit.events (
    event_id        UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID,
    user_id         UUID,
    session_id      TEXT,
    event_type      TEXT NOT NULL,  -- 'message.created', 'document.ingested', 'tool.executed'
    resource_type   TEXT,
    resource_id     UUID,
    action          TEXT NOT NULL,  -- 'create', 'read', 'update', 'delete', 'execute'
    outcome         TEXT NOT NULL CHECK (outcome IN ('success', 'failure', 'partial')),
    ip_address      INET,
    user_agent      TEXT,
    request_id      TEXT,
    details         JSONB NOT NULL DEFAULT '{}',
    error_code      TEXT,
    error_message   TEXT,
    duration_ms     INTEGER,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now()
) PARTITION BY RANGE (created_at);

-- Create monthly partitions
CREATE TABLE audit.events_2025_01 PARTITION OF audit.events
    FOR VALUES FROM ('2025-01-01') TO ('2025-02-01');
CREATE TABLE audit.events_2025_02 PARTITION OF audit.events
    FOR VALUES FROM ('2025-02-01') TO ('2025-03-01');
-- ... additional partitions managed by pg_partman in production

CREATE INDEX idx_audit_tenant_time ON audit.events(tenant_id, created_at DESC);
CREATE INDEX idx_audit_user_time ON audit.events(user_id, created_at DESC);
CREATE INDEX idx_audit_event_type ON audit.events(event_type, created_at DESC);
CREATE INDEX idx_audit_resource ON audit.events(resource_type, resource_id);

-- ============================================================
-- OPERATIONS TABLES
-- ============================================================
CREATE TABLE ops.ingestion_jobs (
    job_id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID NOT NULL REFERENCES ai.tenants(tenant_id),
    collection_id   UUID NOT NULL REFERENCES ai.document_collections(collection_id),
    document_id     UUID REFERENCES ai.documents(document_id),
    celery_task_id  TEXT,
    job_type        TEXT NOT NULL CHECK (job_type IN ('ingest', 're_embed', 'delete', 'summarize')),
    status          TEXT NOT NULL DEFAULT 'queued'
                        CHECK (status IN ('queued', 'running', 'done', 'failed', 'retrying')),
    attempts        INTEGER NOT NULL DEFAULT 0,
    max_attempts    INTEGER NOT NULL DEFAULT 3,
    error           TEXT,
    progress        JSONB NOT NULL DEFAULT '{"current": 0, "total": 0}',
    enqueued_at     TIMESTAMPTZ NOT NULL DEFAULT now(),
    started_at      TIMESTAMPTZ,
    completed_at    TIMESTAMPTZ
);

CREATE INDEX idx_jobs_status ON ops.ingestion_jobs(status, enqueued_at);
CREATE INDEX idx_jobs_tenant ON ops.ingestion_jobs(tenant_id);

-- Application role sees only its tenant's data
CREATE ROLE app_user;

CREATE POLICY tenant_isolation_documents ON ai.documents
    FOR ALL TO app_user
    USING (tenant_id = current_setting('app.tenant_id')::UUID);

CREATE POLICY tenant_isolation_chunks ON ai.document_chunks
    FOR ALL TO app_user
    USING (tenant_id = current_setting('app.tenant_id')::UUID);

CREATE POLICY tenant_isolation_conversations ON ai.conversations
    FOR ALL TO app_user
    USING (tenant_id = current_setting('app.tenant_id')::UUID);

CREATE POLICY tenant_isolation_messages ON ai.messages
    FOR ALL TO app_user
    USING (tenant_id = current_setting('app.tenant_id')::UUID);

-- Users can only see their own conversations (admins see all in tenant)
CREATE POLICY user_conversation_access ON ai.conversations
    FOR SELECT TO app_user
    USING (
        tenant_id = current_setting('app.tenant_id')::UUID
        AND (
            user_id = current_setting('app.user_id')::UUID
            OR EXISTS (
                SELECT 1 FROM ai.users u
                WHERE u.user_id = current_setting('app.user_id')::UUID
                AND u.role = 'admin'
                AND u.tenant_id = current_setting('app.tenant_id')::UUID
            )
        )
    );
